**Part B — Fix PARCIAL collection rate calculation:**

The current collection rate formula (around line 323-325) is:
```ts
const collectionRate = installmentsDueToDate > 0
  ? (installmentsPaidToDate / installmentsDueToDate) * 100
  : 0;
```

This only counts fully PAGADA installments. PARCIAL installments (status = "PARCIAL", meaning partial payment has been made) are counted as zero — they appear in `installmentsOverdue` instead.

**Fix approach:** Add a new query to the `Promise.all` block that fetches PARCIAL installments due to date, with their `amount` and `paidAmount` fields. Then compute proportional credit.

Add this query AFTER the `installmentsPaidToDate` count query (insert as a new entry in the `Promise.all` array):

```ts
// Partial installments due to date (PARCIAL status, with paidAmount for proportional credit)
prisma.installment.findMany({
  where: {
    dueDate: { gte: startOfYear, lte: endOfYear < today ? endOfYear : today },
    status: "PARCIAL",
    ...(developmentId
      ? { sale: { lot: { developmentId } } }
      : {}),
  },
  select: {
    amount: true,
    paidAmount: true,
  },
}),
```

Update the destructuring to include the new result. Name it `partialInstallmentsDue`. Add it after `installmentsPaidToDate` in the destructuring array.

Now compute the proportional credit from PARCIAL installments:

```ts
// Proportional credit from PARCIAL installments
const partialCredit = partialInstallmentsDue.reduce((acc, inst) => {
  const amount = Number(inst.amount ?? 0);
  const paid = Number(inst.paidAmount ?? 0);
  return acc + (amount > 0 ? paid / amount : 0);
}, 0);
```

Update the collection rate formula to include partial credit:

```ts
const collectionRate = installmentsDueToDate > 0
  ? ((installmentsPaidToDate + partialCredit) / installmentsDueToDate) * 100
  : 0;
```

interface MonthlyRow {
  month: number;
  label: string;

  usdExpense: number;

const MONTH_LABELS = [

  "Julio", "Agosto", "Septiembre", "Octubre", "Noviembre", "Diciembre",
];

export function MonthlyMovementsTable({ monthlyData, movementsByType, selectedYear }: Props) {
  const [selectedType, setSelectedType] = useState<string>("all");

  // Determine which types are present in the data
  const availableTypes = useMemo(() => {
    const types = new Set<string>();
    for (const m of movementsByType) {
      types.add(m.type);
    }
    return Array.from(types).sort();
  }, [movementsByType]);

  // Compute filtered monthly data based on selected type
  const filteredMonthlyData = useMemo(() => {
    if (selectedType === "all") {
      return monthlyData;
    }

    // Aggregate movementsByType for the selected type
    const monthMap = new Map<number, MonthlyRow>();
    for (let m = 0; m < 12; m++) {
      monthMap.set(m, {
        month: m,
        label: MONTH_LABELS[m],
        usdIncome: 0,
        usdExpense: 0,
        arsIncome: 0,
        arsExpense: 0,
        netUsd: 0,
      });
    }

    for (const entry of movementsByType) {
      if (entry.type !== selectedType) continue;
      const row = monthMap.get(entry.month)!;
      row.usdIncome += entry.usdIncome;
      row.usdExpense += entry.usdExpense;
      row.arsIncome += entry.arsIncome;
      row.arsExpense += entry.arsExpense;
    }

    return Array.from(monthMap.values()).map((row) => ({
      ...row,
      netUsd: row.usdIncome - row.usdExpense,
    }));

  // Find max USD income for bar width calculation
